AMM算法（Adleman-Mander-Miller Method）

参考资料[Adleman-Mander-Miller Method.pdf](https://github.com/allegory1024/documents/blob/main/Adleman-Mander-Miller Method.pdf)

先开个平方
MATHJAX-SSR-0
贴个代码

import random

def exp(base, power):
    left = 0
    right = power // base
    mid = (left + right) // 2
    while left + 1 < right :
        if power % pow(base, mid) == 0:
            left = mid
        else:
            right = mid
        mid = (left + right) // 2
    return mid

def AMM2(x,p):
    # 万一是二次非剩余
    if pow(x, (p - 1) // 2, p) != 1:
        return -1
    # 生成二次非剩余
    y = random.randint(1, p)
    while pow(y, (p - 1) // 2, p) == 1:
        y = random.randint(1, p)
    # 求t,s
    t = exp(2, p - 1)
    s = (p - 1) // pow(2, t)
    a = pow(y, s, p)
    b = pow(x, s, p)
    h = 1
    for i in range(1, t):
        tmp = pow(2, t - 1 - i, p)
        d = pow(b, tmp, p)
        if d == 1:
            k = 0
        else :
            k = 1
        b = b * pow(a * a, k, p)
        h = h * pow(a, k, p)
        a = a * a
    return pow(x, (s + 1) // 2, p) * h % p
