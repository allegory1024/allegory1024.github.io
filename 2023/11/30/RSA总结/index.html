def exgcd(a, b):     
    if b == 0:         
        return 1, 0    
    else:         
        x, y = exgcd(b, a % b)        
        x, y = y, (x - (a // b) * y)         
        return x, y

from Crypto.Util.number import *

pbits = 1024
ebits = 128
# 获取公钥私钥
p, q, r = getPrime(pbits), getPrime(pbits), getPrime(pbits)
n = p * q * r
phi = (p - 1) * (q - 1) * (r - 1)
# 如果e和phi公因数不为1，两个不同的素数一定是互质的，则无法找到唯一逆元
e = getPrime(ebits)
d = pow(e, -1, phi)
flag = b'Aurora{qsdzyyds!!!}'
m = bytes_to_long(flag)
c = pow(m, d, n)
# print(f'{p = }')
# print(f'{q = }')
# print(f'{r = }')
# print(f'{d = }')
# print(f'{c = }')
print(long_to_bytes(pow(c,pow(d, -1, phi),n)))
