乱入的二分开方

算法

def sqrt_binary(n):
    low = 0
    up = n
    a = 1
    while True:
        mid = (low + up) // 2
        if ABS(mid*mid - n) < a:
            return mid
        else:
            if mid * mid > n:
                up = mid
            if mid * mid < n:
                low = mid

def exgcd(a, b):     
    if b == 0:         
        return 1 , 0
    else:         
        x, y = exgcd(b, a % b)        
        x, y = y, (x - (a // b) * y)         
        return x , y
