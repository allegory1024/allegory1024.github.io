def sqrt_binary(n):
    low = 0
    up = n
    a = 1
    while True:
        mid = (low + up) // 2
        if ABS(mid*mid - n) < a:
            return mid
        else:
            if mid * mid > n:
                up = mid
            if mid * mid < n:
                low = mid
